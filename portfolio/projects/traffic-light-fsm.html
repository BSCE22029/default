Verilog Implementation

// Traffic Light FSM — Verilog HDL
// 4-state Moore FSM, two-way intersection (NS / EW)
// Author: Muhammad Moiz Ahmad

module traffic_light_fsm (
    input       clk, rst,
    output reg [1:0] ns_light,  // 00=Red 01=Yellow 10=Green
    output reg [1:0] ew_light
);

parameter S0 = 2'b00;  // NS Green, EW Red
parameter S1 = 2'b01;  // NS Yellow, EW Red
parameter S2 = 2'b10;  // NS Red, EW Green
parameter S3 = 2'b11;  // NS Red, EW Yellow

reg [1:0] state;
reg [3:0] timer;

always @(posedge clk or posedge rst) begin
    if (rst) begin
        state <= S0; timer <= 4'd9;
    end else if (timer == 0) begin
        case (state)
            S0: begin state <= S1; timer <= 4'd2; end
            S1: begin state <= S2; timer <= 4'd9; end
            S2: begin state <= S3; timer <= 4'd2; end
            S3: begin state <= S0; timer <= 4'd9; end
        endcase
    end else
        timer <= timer - 1;
end

always @(*) begin
    case (state)
        S0: begin ns_light=2'b10; ew_light=2'b00; end
        S1: begin ns_light=2'b01; ew_light=2'b00; end
        S2: begin ns_light=2'b00; ew_light=2'b10; end
        S3: begin ns_light=2'b00; ew_light=2'b01; end
    endcase
end
endmodule
