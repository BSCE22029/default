// Smart Temperature Monitor
// Arduino Nano + DHT11 + Serial logging
// Author: Muhammad Moiz Ahmad

#include <DHT.h>
#include <Wire.h>
#include <Adafruit_SSD1306.h>

#define DHT_PIN     2
#define DHT_TYPE    DHT11
#define ALERT_TEMP  35.0
#define ALERT_HUM   80.0
#define BUZZER_PIN  9
#define LED_PIN     13

DHT dht(DHT_PIN, DHT_TYPE);
Adafruit_SSD1306 display(128, 32, &Wire);

void setup() {
  Serial.begin(115200);
  dht.begin();
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(LED_PIN, OUTPUT);
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  Serial.println("[BOOT] Smart Temp Monitor v1.0");
}

void loop() {
  float temp = dht.readTemperature();
  float hum  = dht.readHumidity();
  float hi   = dht.computeHeatIndex(temp, hum, false);

  if (isnan(temp) || isnan(hum)) {
    Serial.println("[ERR] DHT11 read failed");
    return;
  }

  Serial.print("[DATA] T="); Serial.print(temp);
  Serial.print("°C  H=");  Serial.print(hum);
  Serial.print("%  HI="); Serial.println(hi);

  bool alert = (temp > ALERT_TEMP || hum > ALERT_HUM);
  digitalWrite(LED_PIN, alert);
  if (alert) {
    tone(BUZZER_PIN, 1000, 200);
    Serial.println("[WARN] Threshold exceeded!");
  }

  updateDisplay(temp, hum, hi);
  delay(2000);
}
