Verilog Implementation

// 4-bit ALU — Verilog HDL
// Author: Muhammad Moiz Ahmad

module alu_4bit (
    input  [3:0] A, B,
    input  [2:0] op,
    output reg [3:0] result,
    output reg        carry, zero, negative, overflow
);

reg [4:0] tmp;

always @(*) begin
    tmp = 5'b0;
    case (op)
        3'b000: tmp = A + B;          // ADD
        3'b001: tmp = A - B;          // SUB
        3'b010: tmp = A & B;          // AND
        3'b011: tmp = A | B;          // OR
        3'b100: tmp = A ^ B;          // XOR
        3'b101: tmp = ~A;             // NOT
        3'b110: tmp = A << 1;        // SHL
        3'b111: tmp = A >> 1;        // SHR
    endcase
    result   = tmp[3:0];
    carry    = tmp[4];
    zero     = (result == 4'b0);
    negative = result[3];
    overflow = (op==3'b000) ? (A[3]==B[3] && result[3]!=A[3]) : 1'b0;
end
endmodule
